// Javascript Markov Chain & Hot-Cold Prediction Algorithm
class LotteryPredictor {
    constructor(drawsHistory) {
        this.history = drawsHistory; // Array of draws { numbers: [...], bonus: [...] }
    }

    // 1. Calculate weights based on delays
    calculateDelayWeights(maxNum) {
        const delays = Array(maxNum + 1).fill(0);
        const weights = Array(maxNum + 1).fill(1.0);
        
        for (let num = 1; num <= maxNum; num++) {
            const lastIdx = this.history.findIndex(h => h.numbers.includes(num));
            delays[num] = lastIdx === -1 ? this.history.length : lastIdx;
            // Higher delay = higher probability weight (mean reversion)
            weights[num] += Math.log(delays[num] + 1) * 1.5;
        }
        return weights;
    }

    // 2. Markov Chain transition probability matrix (First-order)
    calculateMarkovTransitions(maxNum) {
        const matrix = Array(maxNum + 1).fill(0).map(() => Array(maxNum + 1).fill(0));
        
        for (let idx = 0; idx < this.history.length - 1; idx++) {
            const currentDraw = this.history[idx].numbers;
            const nextDraw = this.history[idx + 1].numbers;
            
            currentDraw.forEach(c => {
                nextDraw.forEach(n => {
                    matrix[c][n]++;
                });
            });
        }
        return matrix;
    }

    // 3. Generate Top Predictions
    predict(maxNum, count, bonusMax = 0) {
        const delayWeights = this.calculateDelayWeights(maxNum);
        const transitionMatrix = this.calculateMarkovTransitions(maxNum);
        
        const finalScores = Array(maxNum + 1).fill(0);
        const lastDraw = this.history[0]?.numbers || [];

        for (let num = 1; num <= maxNum; num++) {
            let score = delayWeights[num];
            // Add transition weights from the last drawn numbers
            lastDraw.forEach(lastNum => {
                score += (transitionMatrix[lastNum][num] || 0) * 2.0;
            });
            finalScores[num] = score;
        }

        // Sort by final prediction score
        const predictions = [];
        for (let i = 1; i <= maxNum; i++) {
            predictions.push({ number: i, score: finalScores[i] });
        }
        predictions.sort((a, b) => b.score - a.score);
        
        const mainNumbers = predictions.slice(0, count).map(p => p.number).sort((a,b)=>a-b);
        
        // Bonus predictions if applicable
        const bonusNumbers = [];
        if (bonusMax > 0) {
            const bonusFreq = {};
            this.history.forEach(d => {
                d.bonus.forEach(b => {
                    bonusFreq[b] = (bonusFreq[b] || 0) + 1;
                });
            });
            const bSorted = Object.keys(bonusFreq).sort((x, y) => bonusFreq[y] - bonusFreq[x]).map(Number);
            bonusNumbers.push(bSorted[0] || 1);
        }
        
        return { numbers: mainNumbers, bonus: bonusNumbers };
    }
}
# Python Predictive Engine using Pandas & Scikit-Learn Regression
import numpy as np
import pandas as pd
from sklearn.linear_model import PoissonRegressor
from sklearn.preprocessing import StandardScaler

class OPAPLotteryMLPredictor:
    def __init__(self, draws_dataframe):
        self.df = draws_dataframe # columns: ['draw_id', 'date', 'n1', 'n2', 'n3', 'n4', 'n5', 'bonus']

    def engineer_features(self, max_number):
        features = []
        targets = []
        
        # Calculate historical stats prior to each draw for features
        for idx in range(50, len(self.df)):
            sub_df = self.df.iloc[idx:] # historical draws prior to current
            
            # Feature calculation: Delays & rolling frequency
            for num in range(1, max_number + 1):
                delay = 0
                freq_50 = 0
                for _, row in sub_df.iterrows():
                    draw_nums = [row['n1'], row['n2'], row['n3'], row['n4'], row['n5']]
                    if num in draw_nums:
                        break
                    delay += 1
                
                for _, row in sub_df.head(50).iterrows():
                    draw_nums = [row['n1'], row['n2'], row['n3'], row['n4'], row['n5']]
                    if num in draw_nums:
                        freq_50 += 1
                        
                features.append([delay, freq_50])
                
                # Target: 1 if number is drawn in current draw, 0 otherwise
                current_draw_nums = [self.df.iloc[idx]['n1'], self.df.iloc[idx]['n2'], self.df.iloc[idx]['n3'], self.df.iloc[idx]['n4'], self.df.iloc[idx]['n5']]
                targets.append(1 if num in current_draw_nums else 0)
                
        return np.array(features), np.array(targets)

    def train_and_predict(self, max_number, select_count):
        X, y = self.engineer_features(max_number)
        
        # Scale features
        scaler = StandardScaler()
        X_scaled = scaler.fit_transform(X)
        
        # Fit Poisson Regression model
        model = PoissonRegressor(alpha=1.0)
        model.fit(X_scaled, y)
        
        # Calculate features for current state (for next draw)
        live_features = []
        for num in range(1, max_number + 1):
            delay = 0
            freq_50 = 0
            for _, row in self.df.iterrows():
                draw_nums = [row['n1'], row['n2'], row['n3'], row['n4'], row['n5']]
                if num in draw_nums:
                    break
                delay += 1
            
            for _, row in self.df.head(50).iterrows():
                draw_nums = [row['n1'], row['n2'], row['n3'], row['n4'], row['n5']]
                if num in draw_nums:
                    freq_50 += 1
            live_features.append([delay, freq_50])
            
        live_features_scaled = scaler.transform(np.array(live_features))
        probabilities = model.predict(live_features_scaled)
        
        # Sort and select top candidates
        top_indices = np.argsort(probabilities)[::-1][:select_count]
        predicted_numbers = [i + 1 for i in top_indices]
        predicted_numbers.sort()
        
        return predicted_numbers
